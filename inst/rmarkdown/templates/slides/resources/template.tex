      \usebeamerfont{title}\inserttitle\par%
      \ifx\insertsubtitle\@empty%
      \else%
        \vskip0.25em%
        {\usebeamerfont{subtitle}\usebeamercolor[fg]{subtitle}\insertsubtitle\par}%
      \fi%

    \begin{beamercolorbox}[sep=8pt,center]{author}

% Footer with NMU coloured lines and slide numbers (based on original template)
\defbeamertemplate*{footline}{nmu footline}{%
    \ifnum\insertpagenumber=1
    \hbox{%
        \begin{beamercolorbox}[wd=\paperwidth,ht=.8ex,dp=1ex,center]{}%
      % empty environment for title page
        \end{beamercolorbox}%
    }%
    \vskip0pt%
    \else%
        \Tiny{%
            \hfill%
            \vspace*{1pt}%
            \insertframenumber/\inserttotalframenumber \hspace*{0.1cm}%
            \newline%
            \color{nmublue}{\rule{\paperwidth}{0.4mm}}\newline%
            \color{nmuyellow}{\rule{\paperwidth}{0.4mm}}%
        }%
    \fi%
}
